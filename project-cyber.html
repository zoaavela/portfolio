import random
import string

def generer_mot_de_passe(longueur, inclure_symboles):
    """
    Génère une chaîne aléatoire robuste basée sur les critères.
    """
    caracteres = string.ascii_letters + string.digits
    
    if inclure_symboles:
        caracteres += "!@#$%^&*"

    return "".join(random.choice(caracteres) for _ in range(longueur))

def lancer_programme():
    print("--- GÉNÉRATEUR DE MOTS DE PASSE ---")

    while True:
        try:
            longueur = int(input("Longueur du mot de passe (min 4) : "))
            if longueur >= 4:
                break
            print("Erreur : La longueur doit être d'au moins 4.")
        except ValueError:
            print("Erreur : Veuillez entrer un nombre valide.")

    reponse = input("Inclure des symboles spéciaux ? (o/n) : ").lower()
    avec_symboles = (reponse == 'o')

    pwd = generer_mot_de_passe(longueur, avec_symboles)

    print("\n" + "-" * 30)
    print(f"Résultat : {pwd}")
    print("-" * 30)

if __name__ == "__main__":
    lancer_programme()
