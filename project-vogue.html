import os

# Configuration du jeu
QUESTIONS = [
    {
        "question": "Quelle est la capitale de la France ?",
        "options": ["A. Lyon", "B. Marseille", "C. Paris"],
        "reponse": "C"
    },
    {
        "question": "Combien font 5 x 5 ?",
        "options": ["A. 10", "B. 25", "C. 55"],
        "reponse": "B"
    },
    {
        "question": "Quel langage utilise-t-on pour le Web (Front-end) ?",
        "options": ["A. Python", "B. JavaScript", "C. C++"],
        "reponse": "B"
    }
]

FICHIER_SCORE = "meilleur_score.txt"

def charger_meilleur_score():
    """Lit le fichier pour trouver le record actuel."""
    if os.path.exists(FICHIER_SCORE):
        with open(FICHIER_SCORE, "r") as f:
            try:
                return int(f.read())
            except ValueError:
                return 0
    return 0

def sauvegarder_score(nouveau_score):
    """Ecrase le fichier avec le nouveau record."""
    with open(FICHIER_SCORE, "w") as f:
        f.write(str(nouveau_score))

def lancer_jeu():
    score = 0
    record_actuel = charger_meilleur_score()

    print("\n--- BIENVENUE DANS LE QUIZ MASTER ---")
    print(f"Record à battre : {record_actuel} points\n")

    for i, q in enumerate(QUESTIONS):
        print(f"Question {i + 1} : {q['question']}")
        
        for option in q['options']:
            print(option)
        
        reponse_user = input("Votre réponse (A, B ou C) : ").upper()

        if reponse_user == q['reponse']:
            print("Bonne réponse !\n")
            score += 1
        else:
            print(f"Mauvaise réponse. C'était la {q['reponse']}.\n")

    print(f"--- TERMINÉ ! VOTRE SCORE : {score}/{len(QUESTIONS)} ---")

    # Mise à jour du record si nécessaire
    if score > record_actuel:
        print("INCROYABLE ! NOUVEAU RECORD ENREGISTRÉ !")
        sauvegarder_score(score)
    else:
        print(f"Dommage, le record reste à {record_actuel}.")

if __name__ == "__main__":
    lancer_jeu()
